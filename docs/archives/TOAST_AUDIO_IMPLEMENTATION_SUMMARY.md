## 🚀 مثال الاستخدام الأساسي

```tsx
import { useToast } from '@/context/ToastContext';

export function MyComponent() {
  const toast = useToast();

  // 1️⃣ رسالة نجاح
  const handleSave = () => {
    toast.success('تم حفظ البيانات بنجاح!', 'حفظ');
  };

  // 2️⃣ رسالة خطأ
  const handleError = () => {
    toast.error('فشل الاتصال بالخادم', 'خطأ');
  };

  // 3️⃣ حوار تأكيد (حذف)
  const handleDelete = async (id: string, name: string) => {
    const confirmed = await toast.confirm({
      title: 'حذف العنصر',
      message: `سيتم حذف "${name}" بشكل نهائي`,
      confirmText: 'نعم، احذف',
      cancelText: 'إلغاء',
      isDangerous: true,
      onConfirm: async () => {
        // تنفيذ الحذف
        toast.delete(`تم حذف "${name}"`, 'تم الحذف');
      }
    });
  };

  return (
    <div>
      <button onClick={handleSave}>حفظ</button>
      <button onClick={handleError}>خطأ</button>
      <button onClick={() => handleDelete('1', 'العقد')}>حذف</button>
    </div>
  );
}
```
